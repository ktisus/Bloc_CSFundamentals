function square(arr) {
  
 var sum = 0;
 
  
  for (var i=0; i< arr.length; i++) {
    
    sum += arr[i]; 
  
  }
    var n = Math.sqrt(sum);
  
   if (n % 1 ===0) {
    return n;
   }
   else {
     return sum;
   }
 }

function duplicate(arr) {
  
 for (var i=0; i< arr.length; i++) {
    
     for (var j=0; j<i; j++){
        
       if (arr[j] === arr[i]) {
          return true;
        }
     }
  
  }
   return false; 
 }

function anagramWord (a, b) {
  x = a.split('').sort().join('');
  y = b.split('').sort().join('')
  
  if (x === y) {
    return "An anagram"
  }
  else 
    return "not anagram"
}
